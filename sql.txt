创建house表
CREATE TABLE house (
hid INTEGER GENERATED BY DEFAULT AS IDENTITY,
city varchar(50),
zone varchar(50),
room varchar(50) NOT NULL,
area integer not null,
mode varchar(100) NOT NULL,
rent float not null,
description varchar(200) NOT NULL,
phone varchar(12),
PRIMARY KEY (hid),
state integer
)

插入数据insert into house(city,zone,room,area,mode,rent,description,phone,state)
